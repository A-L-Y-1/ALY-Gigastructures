			factor = 1.5

		}
		modifier = {
			factor = 1.5

				has_trait = trait_robust

			}

			factor = 0.8
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
		}

			factor = 1.5

			factor = 0.85

	}

	possible_pre_triggers = {
		has_owner = yes

	possible = { NOT = { has_trait = trait_nerve_stapled } }

			unity = 2
		}

		modifier = {
			factor = 1.5
			has_job = giga_megaengineer_drone

			factor = 0.5

		}
	}
}
