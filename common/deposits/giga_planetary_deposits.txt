d_birch_lost01 = {
	time = 120
	is_for_colonizeable = yes
	category = deposit_cat_blockers
	potential = { always = no }

	resources = {
		category = deposit_blockers
		cost = { energy = 2500 }
	}

	on_cleared = {
		add_deposit = d_birch_lost02
		if = { limit = { any_owned_pop = { is_gestalt = no is_robot_pop = no } }
			random_owned_pop = {
				limit = { is_gestalt = no is_robot_pop = no }
				root = {
					create_pop = { species = prev }
					create_pop = { species = prev }
				}
			}
		}
		else = {
			random_owned_pop = {
				root = {
					create_pop = { species = prev }
					create_pop = { species = prev }
				}
			}
		}
	}
}
