# Origin Birch Restoration

giga_origin_birch_restore = {
	owned_planets_only = yes
	icon = decision_birch_world

	enactment_time = 1800
	resources = {
		category = decisions
		cost = {
			alloys = 40000
			unity = 5000
			influence = 250
		}
	}

	potential = {
		is_planet_class = pc_origin_birch
		has_planet_flag = birch_expedition_10_complete
		exists = owner
		owner = {
			has_origin = origin_birch
		}
	}

	abort_trigger = {
		or = {
			not = { exists = owner }
			owner = {
				not = {
					has_origin = origin_birch
				}
			}
		}
	}

	effect = {
		tooltip = {
			change_pc = pc_birch
		}
		custom_tooltip = giga_origin_birch_restore_tooltip

		hidden_effect = {
			# set flag for speical effects like minimum insulae
			set_planet_flag = giga_origin_birch 

			# gather information for district replacement

			# city districts
			if = {
				limit = {
					owner = { is_gestalt = no }
				}
				export_trigger_value_to_variable = {
					trigger = num_districts
					parameters = {
						type = district_giga_planet_city
					}
					variable = birch_districts_city
				}
			}
			else_if = {
				limit = {
					owner = { is_hive_empire = yes }
				}
				export_trigger_value_to_variable = {
					trigger = num_districts
					parameters = {
						type = district_giga_planet_hive
					}
					variable = birch_districts_city
				}
			}
			else_if = {
				limit = {
					owner = { is_machine_empire = yes }
				}
				export_trigger_value_to_variable = {
					trigger = num_districts
					parameters = {
						type = district_giga_planet_nexus
					}
					variable = birch_districts_city
				}
			}
			
			# industrial districts
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = {
					type = district_industrial
				}
				variable = birch_districts_industrial
			}

			# generator districts
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = {
					type = district_giga_planet_generator
				}
				variable = birch_districts_generator
			}

			# mining districts
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = {
					type = district_giga_planet_mining
				}
				variable = birch_districts_mining
			}

			# farming districts
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = {
					type = district_giga_planet_farming
				}
				variable = birch_districts_farming
			}

			# research districts
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = {
					type = district_giga_origin_birch_lab
				}
				variable = birch_districts_research
			}


			# apply conversion modifier
			add_modifier = {
				modifier = frameworld_building_wreck_fix # TODO proper modifier
				days = 5
			}

			# change planet class
			change_pc = pc_birch
			add_deposit = d_giga_birch_mechanics
			remove_all_districts = yes

			# fill in new districts
			add_district = district_giga_birch_structure
			add_district = district_giga_birch_orykta
			add_district = district_giga_birch_ktisma
			add_district = district_giga_birch_physma


			# replace features
			

			# notification
			planet_event = {
				id = giga_birch.10001
			}
		}
	}
}
