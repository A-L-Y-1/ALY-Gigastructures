hrae_mc_0 = {
	entity = "giga_consite_10_10_entity"
	construction_entity = "giga_consite_10_10_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	prerequisites = { giga_tech_hrae_mc }

	build_time = 1800
	resources = {
		category = megastructures
		cost = {
			alloys = 5000
			influence = 150
		}
		upkeep = { energy = 5 }
	}

	potential = {
		has_ascension_perk = ap_gigastructural_constructs
		has_technology = giga_tech_hrae_mc
		NOT = { has_global_flag = hraemc_disabled }
	}

	possible = {
		custom_tooltip = { fail_text = "requires_inside_border"								is_inside_border = from }
		custom_tooltip = { fail_text = "requires_surveyed_system"							giga_system_is_surveyed = yes }
		custom_tooltip = { fail_text = "requires_no_existing_megastructure"					giga_has_megastructure = no }
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = { fail_text = "requires_no_existing_megastructure"				giga_planet_has_megastructure = no }
			custom_tooltip = { fail_text = "requires_no_anomaly"							has_anomaly = no }
			custom_tooltip = { fail_text = "must_build_around_black_hole"					is_planet_class = pc_black_hole }
		}
	}

	ai_weight = {
		weight = 100	# Default = 100
		modifier = { factor = 5		from = { has_monthly_income = { resource = energy value < 200 } } }
		modifier = { factor = 0.5	from = { has_monthly_income = { resource = energy value >= 200 } } }
		modifier = { factor = 5		from = { has_monthly_income = { resource = sr_dark_matter value < 10 } } }
		modifier = { factor = 0.5	from = { has_monthly_income = { resource = sr_dark_matter value >= 10 } } }
		modifier = { factor = 6		from = { has_monthly_income = { resource = physics_research value < 200 } } years_passed <= 50 }
		modifier = { factor = 5		from = { has_monthly_income = { resource = physics_research value < 400 } } years_passed <= 100 }
		modifier = { factor = 4		from = { has_monthly_income = { resource = physics_research value < 600 } } years_passed <= 150 }
		modifier = { factor = 3		from = { has_monthly_income = { resource = physics_research value < 800 } } years_passed <= 200 }
		modifier = { factor = 2		from = { has_monthly_income = { resource = physics_research value < 1000 } } years_passed > 200 }
		modifier = { factor = 0.5	from = { has_monthly_income = { resource = physics_research value >= 200 } } years_passed <= 50 }
		modifier = { factor = 0.4	from = { has_monthly_income = { resource = physics_research value >= 400 } } years_passed <= 100 }
		modifier = { factor = 0.3	from = { has_monthly_income = { resource = physics_research value >= 600 } } years_passed <= 150 }
		modifier = { factor = 0.2	from = { has_monthly_income = { resource = physics_research value >= 800 } } years_passed <= 200 }
		modifier = { factor = 0.1	from = { has_monthly_income = { resource = physics_research value >= 1000 } } years_passed > 200 }
		modifier = { factor = 0.0	from = { count_owned_megastructure = { limit = { is_megastructure_type = hrae_mc_4 } count = 1 } } }
		modifier = {
			factor = 0
			from = {
				OR = {
					has_megastructure = hrae_mc_0
					has_megastructure = hrae_mc_1
					has_megastructure = hrae_mc_2
					has_megastructure = hrae_mc_3
				}
			}
		}
	}

	on_build_complete = {
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }
		from = { country_event = { id = giga_dialog.3101 } }	# Notification
		fromfrom.planet = {
			if = { limit = { exists = orbital_station } orbital_station = { dismantle = yes } }
			set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
			set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
		}
	}
}
