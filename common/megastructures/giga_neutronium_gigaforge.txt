neutronium_gigaforge_0 = {
	entity = "giga_consite_7_7_entity"
	construction_entity = "giga_consite_7_7_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	prerequisites = { giga_tech_neutronium_gigaforge }

	build_time = 1800
	resources = {
		category = megastructures
		cost = {
			alloys = 3500
			influence = 75
		}
		upkeep = { energy = 5 }
	}

	potential = {
		has_technology = giga_tech_neutronium_gigaforge
		NOT = { has_global_flag = gigaforge_disabled }
	}

	possible = {
		custom_tooltip = { fail_text = "requires_inside_border"				is_inside_border = from }
		custom_tooltip = { fail_text = "requires_surveyed_system"			NOT = { any_system_planet = { is_surveyed = { who = prev.from status = no } } } }
		custom_tooltip = { fail_text = "requires_no_existing_megastructure"	giga_has_megastructure = no }
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = { fail_text = "requires_no_existing_megastructure"		giga_planet_has_megastructure = no }
			custom_tooltip = { fail_text = "requires_no_anomaly"					has_anomaly = no }
			custom_tooltip = { fail_text = "must_build_around_neutron_star"			OR = { is_planet_class = pc_neutron_star is_planet_class = pc_pulsar } }
		}
	}

	ai_weight = {
		weight = 100	# Default = 100
		modifier = { factor = 10	from = { has_monthly_income = { resource = alloys value < 200 } } }
		modifier = { factor = 0.1	from = { has_monthly_income = { resource = energy value < 50 } } }
		modifier = { factor = 6		from = { has_monthly_income = { resource = engineering_research value < 200 } } years_passed <= 50 }
		modifier = { factor = 5		from = { has_monthly_income = { resource = engineering_research value < 400 } } years_passed <= 100 }
		modifier = { factor = 4		from = { has_monthly_income = { resource = engineering_research value < 600 } } years_passed <= 150 }
		modifier = { factor = 3		from = { has_monthly_income = { resource = engineering_research value < 800 } } years_passed <= 200 }
		modifier = { factor = 2		from = { has_monthly_income = { resource = engineering_research value < 1000 } } years_passed > 200 }
		modifier = { factor = 0.5	from = { has_monthly_income = { resource = engineering_research value >= 200 } } years_passed <= 50 }
		modifier = { factor = 0.4	from = { has_monthly_income = { resource = engineering_research value >= 400 } } years_passed <= 100 }
		modifier = { factor = 0.3	from = { has_monthly_income = { resource = engineering_research value >= 600 } } years_passed <= 150 }
		modifier = { factor = 0.2	from = { has_monthly_income = { resource = engineering_research value >= 800 } } years_passed <= 200 }
		modifier = { factor = 0.1	from = { has_monthly_income = { resource = engineering_research value >= 1000 } } years_passed > 200 }
		modifier = { factor = 0.05	from = { count_owned_megastructure = { limit = { OR = { is_megastructure_type = neutronium_gigaforge_3 is_megastructure_type = neutronium_gigaforge_restored } } count > 3 } } }
		modifier = { factor = 0.15	from = { count_owned_megastructure = { limit = { OR = { is_megastructure_type = neutronium_gigaforge_3 is_megastructure_type = neutronium_gigaforge_restored } } count = 3 } } }
		modifier = { factor = 0.20	from = { count_owned_megastructure = { limit = { OR = { is_megastructure_type = neutronium_gigaforge_3 is_megastructure_type = neutronium_gigaforge_restored } } count = 2 } } }
		modifier = { factor = 0.25	from = { count_owned_megastructure = { limit = { OR = { is_megastructure_type = neutronium_gigaforge_3 is_megastructure_type = neutronium_gigaforge_restored } } count = 1 } } }
		modifier = { factor = 0
			from = {
				OR = {
					has_megastructure = neutronium_gigaforge_0
					has_megastructure = neutronium_gigaforge_1
					has_megastructure = neutronium_gigaforge_2
					has_megastructure = neutronium_gigaforge_ruined
				}
			}
		}
	}

	on_build_complete = {
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }
		from = { country_event = { id = giga_dialog.701 } }	# Notification
		fromfrom.planet = {
			if = { limit = { exists = orbital_station } orbital_station = { dismantle = yes } }
			set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
			set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
		}
	}
}
